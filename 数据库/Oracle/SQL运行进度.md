```sql
SELECT A.USERNAME,
       A.SID,
       A.SERIAL#,
       A.OPNAME,
       A.TARGET,
       A.START_TIME,
       A.LAST_UPDATE_TIME,
       C.OSUSER,
       C.MACHINE,
       C.PROGRAM,
       ROUND(A.SOFAR * 100 / TOTALWORK, 0) || '%' AS SCHEDULE_PCT,
       A.TIME_REMAINING,
       A.ELAPSED_SECONDS,
       B.SQL_TEXT
  FROM GV$SESSION_LONGOPS A
  INNER JOIN GV$SQL B
    ON A.SQL_ADDRESS = B.ADDRESS
   AND A.SQL_HASH_VALUE = B.HASH_VALUE
   AND A.INST_ID = B.INST_ID
  INNER JOIN GV$SESSION C
    ON A.SID = C.SID
   AND A.INST_ID = C.INST_ID
   AND C.STATUS = 'ACTIVE'
 
```
